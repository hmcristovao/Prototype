PREFIX dbpedia2: <http://dbpedia.org/property/>

PREFIX skos: <http://www.w3.org/2004/02/skos/core#>

PREFIX relationship: <http://relationship.org/>

   :Concept_map ?predicate ?object .
   :Concept_map relationship:homepage ?homepage . 
   ?subject ?predicate2 :Concept_map .
}
WHERE {
  {
    { 
      :Concept_map ?predicate ?object . 
      FILTER regex(?object, "http://dbpedia.org/resource/")
    }
    UNION
    { 
      :Concept_map foaf:isPrimaryTopicOf ?homepage .   
    }
    UNION
    { 
      ?subject ?predicate2 :Concept_map .
      FILTER regex(?subject, "http://dbpedia.org/resource/")
